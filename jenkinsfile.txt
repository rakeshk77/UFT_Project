        }

            steps {
                echo 'Building project...'

        }

        stage('Archive Results') {
            steps {
                echo 'Archiving UFT test results...'
                archiveArtifacts artifacts: 'C:\\UFTResults\\**', allowEmptyArchive: true
            }
        }

        stage('Deploy') {
            steps {
                echo 'Deploying application...'
                // Add deployment steps if necessary.
